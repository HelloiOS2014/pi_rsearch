Environment:

\`;

  const relevantGuidelines = tools
    .map((t) => TOOL_GUIDELINES[t])

export const codeSessionMigration = `export const CURRENT_SESSION_VERSION = 3;

/** Migrate v1 → v2: add id/parentId tree structure. Mutates in place. */
function migrateV1ToV2(entries: FileEntry[]): void {
  const ids = new Set<string>();
  let prevId: string | null = null;

  for (const entry of entries) {
    if (entry.type === "session") {
      entry.version = 2;
      continue;
    }
    entry.id = generateId(ids);
    entry.parentId = prevId;
    prevId = entry.id;
  }
}

/** Migrate v2 → v3: rename hookMessage role to custom. */
function migrateV2ToV3(entries: FileEntry[]): void {
  for (const entry of entries) {
    if (entry.type === "session") { entry.version = 3; continue; }
    if (entry.type === "message") {
      const msgEntry = entry as SessionMessageEntry;
      if ((msgEntry.message as any).role === "hookMessage") {
        (msgEntry.message as any).role = "custom";
      }
    }
  }
}

/** Run all necessary migrations to bring entries to current version. */
function migrateToCurrentVersion(entries: FileEntry[]): boolean {
  const header = entries.find(e => e.type === "session") as SessionHeader;
  const version = header?.version ?? 1;
  if (version >= CURRENT_SESSION_VERSION) return false;

  if (version < 2) migrateV1ToV2(entries);
  if (version < 3) migrateV2ToV3(entries);
  return true;
}`;

export const codeEstimateTokens = `/**
 * Estimate token count for a message using chars/4 heuristic.
 * This is conservative (overestimates tokens).
 */
export function estimateTokens(message: AgentMessage): number {
  let chars = 0;

  switch (message.role) {
    case "user": {
      const content = message.content;
      if (typeof content === "string") {
        chars = content.length;
      } else if (Array.isArray(content)) {
        for (const block of content) {
          if (block.type === "text" && block.text) {
            chars += block.text.length;
          }
        }
      }
      return Math.ceil(chars / 4);
    }
    case "assistant": {
      for (const block of assistant.content) {
        if (block.type === "text") chars += block.text.length;
        else if (block.type === "thinking") chars += block.thinking.length;
        else if (block.type === "toolCall")
          chars += block.name.length + JSON.stringify(block.arguments).length;
      }
      return Math.ceil(chars / 4);
    }
    case "toolResult": {
      // ...similar logic
      // Images estimated as 4800 chars (≈1200 tokens)
      return Math.ceil(chars / 4);
    }
  }
  return 0;
}`;

export const codeFindValidCutPoints = `/**
 * Find valid cut points: indices of user, assistant, custom, or
 * bashExecution messages.
 * Never cut at tool results (they must follow their tool call).
 */
function findValidCutPoints(
  entries: SessionEntry[],
  startIndex: number,
  endIndex: number,
): number[] {
  const cutPoints: number[] = [];
  for (let i = startIndex; i < endIndex; i++) {
    const entry = entries[i];
    switch (entry.type) {
      case "message": {
        const role = entry.message.role;
        switch (role) {
          case "user":
          case "assistant":
          case "bashExecution":
          case "custom":
          case "branchSummary":
          case "compactionSummary":
            cutPoints.push(i);
            break;
          case "toolResult":
            break;  // NEVER cut here!
        }
        break;
      }
    }
    // branch_summary and custom_message are user-role, valid cut points
    if (entry.type === "branch_summary" || entry.type === "custom_message") {
      cutPoints.push(i);
    }
  }
  return cutPoints;
}`;

export const codeIsRetryableError = `/**
 * Check if an error is retryable (overloaded, rate limit, server errors).
 * Context overflow errors are NOT retryable (handled by compaction instead).
 */
private _isRetryableError(message: AssistantMessage): boolean {
  if (message.stopReason !== "error" || !message.errorMessage) return false;

  // Context overflow is handled by compaction, not retry
  const contextWindow = this.model?.contextWindow ?? 0;
  if (isContextOverflow(message, contextWindow)) return false;

  const err = message.errorMessage;
  return /overloaded|rate.?limit|too many requests|429|500|502|503|504
    |service.?unavailable|connection.?error|fetch failed|timed? out/i.test(err);
}`;

export const codeWithRetry = `export async function withRetry<T>(
  fn: () => Promise<T>,
  options?: RetryOptions,
): Promise<T> {
  const opts = { ...DEFAULT_OPTIONS, ...options };
  let lastError: Error | undefined;

  for (let attempt = 0; attempt <= opts.maxRetries; attempt++) {
    try {
      return await fn();
    } catch (error) {
      lastError = error instanceof Error ? error : new Error(String(error));

      // Don't retry on final attempt or non-retryable errors
      if (attempt === opts.maxRetries || !isRetryable(error)) {
        throw lastError;
      }

      // Exponential backoff + jitter
      const exponentialDelay = opts.baseDelayMs * Math.pow(2, attempt);
      const jitter = Math.random() * opts.baseDelayMs;
      const delay = Math.min(exponentialDelay + jitter, opts.maxDelayMs);

      opts.onRetry?.(lastError, attempt + 1, delay);
      await new Promise(resolve => setTimeout(resolve, delay));
    }
  }
  throw lastError ?? new Error("Retry failed");
}`;

export const codeDeepMergeSettings = `/** Deep merge settings: project/overrides take precedence */
function deepMergeSettings(base: Settings, overrides: Settings): Settings {
  const result: Settings = { ...base };

  for (const key of Object.keys(overrides) as (keyof Settings)[]) {
    const overrideValue = overrides[key];
    const baseValue = base[key];

    if (overrideValue === undefined) continue;

    // For nested objects, merge recursively
    if (
      typeof overrideValue === "object" && overrideValue !== null &&
      !Array.isArray(overrideValue) &&
      typeof baseValue === "object" && baseValue !== null &&
      !Array.isArray(baseValue)
    ) {
      (result as any)[key] = { ...baseValue, ...overrideValue };
    } else {
      // For primitives and arrays, override value wins
      (result as any)[key] = overrideValue;
    }
  }
  return result;
}`;

export const codeAgentPrompt = `async prompt(userMessage: string): Promise<string> {
  this.abortController = new AbortController();
  const messages = this.session.getMessages();
  messages.push({ role: "user", content: userMessage });
  await this.session.append({ role: "user", content: userMessage });

  this.emit({ type: "agent_start", model: this.config.model });

  let turnNumber = 0;
  const MAX_TURNS = 20;

  while (turnNumber < MAX_TURNS) {
    if (this.abortController.signal.aborted) break;
    turnNumber++;
    this.emit({ type: "turn_start", turnNumber });

    // Call the API with retry (Ch.6.4)
    const response = await withRetry(
      () => this.client.messages.create({
        model: this.config.model,
        max_tokens: this.config.maxTokens,
        system: this.systemPrompt,
        tools: ALL_TOOLS as Anthropic.Messages.Tool[],
        messages,
      }),
      { maxRetries: 3, onRetry: (err, attempt, delay) => {
        this.emit({ type: "error", error: new Error(
          \`Retry \${attempt} in \${Math.round(delay)}ms: \${err.message}\`
        )});
      }},
    );

    // Process response content blocks
    let fullText = "";
    const toolUseBlocks: Anthropic.Messages.ToolUseBlock[] = [];

    for (const block of response.content) {
      if (block.type === "text") {
        fullText += block.text;
        this.emit({ type: "text_delta", text: block.text });
      } else if (block.type === "tool_use") {
        toolUseBlocks.push(block);
      }
    }

    // Add assistant message to conversation
    messages.push({ role: "assistant", content: response.content });
    await this.session.append({ role: "assistant", content: response.content });

    // If no tool calls, we're done
    if (toolUseBlocks.length === 0) {
      this.emit({ type: "agent_end", totalTurns: turnNumber });
      return fullText;
    }

    // Execute tool calls (Ch.3)
    const toolResults: Anthropic.Messages.ToolResultBlockParam[] = [];
    for (const toolBlock of toolUseBlocks) {
      if (this.abortController.signal.aborted) break;
      this.emit({ type: "tool_start", name: toolBlock.name, id: toolBlock.id });

      const executor = TOOL_EXECUTORS[toolBlock.name];
      let result: string;
      try {
        result = executor
          ? await executor(toolBlock.input as Record<string, unknown>)
          : \`Unknown tool: \${toolBlock.name}\`;
      } catch (err) {
        result = \`Error: \${err instanceof Error ? err.message : String(err)}\`;
      }

      this.emit({ type: "tool_end", name: toolBlock.name, result });
      toolResults.push({
        type: "tool_result", tool_use_id: toolBlock.id, content: result,
      });
    }

    // Add tool results to conversation
    messages.push({ role: "user", content: toolResults });
    await this.session.append({ role: "user", content: toolResults });
  }

  this.emit({ type: "agent_end", totalTurns: turnNumber });
  return "(max turns reached)";
}`;

### Prompt 的五个区域

`buildSystemPrompt` 按固定顺序拼装 prompt，形成五个清晰的区域：

```
┌────────────────────────────────────────────────┐
│  1. 角色定义 + 工具列表                          │
│     "You are an expert coding assistant..."     │
│     Available tools: read, bash, edit, write    │
├────────────────────────────────────────────────┤
│  2. 自适应 Guidelines                           │
│     根据 selectedTools 动态生成                  │
│     有 bash+grep? → "Prefer grep over bash"    │
│     有 read+edit? → "Read before editing"      │
├────────────────────────────────────────────────┤
│  3. 文档引用                                    │
│     Pi docs/examples/README 路径               │
├────────────────────────────────────────────────┤
│  4. 项目上下文 (Context Files)                  │
│     .pi/instructions.md 等项目级指令            │
│  5. Skills                                     │
```

## 6.3 上下文压缩 (Compaction)

当对话越来越长，最终会撞上模型的 context window 上限。Pi 通过 **Compaction** 机制解决这个问题 —— 把旧消息压缩成摘要，释放空间给新对话。

### Token 估算

压缩的第一步是知道"当前用了多少 token"。Pi 用 `chars / 4` 的启发式方法估算：

为什么用 `chars / 4` 而不是精确的 tokenizer？因为精确 tokenization 需要加载 tokenizer 模型（数 MB），而且不同模型的 tokenizer 不同。`chars / 4` 在英文中非常接近实际值（英文平均 1 token ≈ 4 chars），在中文中会高估（中文 1 字 ≈ 1-2 token），但高估是安全的 —— 最多提前压缩，不会溢出。

生产级 Agent 不能因为一个 API 503 就崩溃。Pi 实现了多层可靠性保障。

### 自动重试 (指数退避)

**错误分类**是关键。Pi 把错误分为三类：

| 错误类型 | 处理方式 | 举例 |
|---------|---------|------|
| **Retryable** | 指数退避重试 | 429, 503, overloaded, rate limit |
| **Context overflow** | 触发 compaction → 重试 | context too long, max tokens exceeded |
| **Fatal** | 显示错误，不重试 | 401, 403, invalid_api_key |

Pi 的 `AgentSession` 在此基础上增加了一个重要特性：**成功即重置**。每次收到成功的 assistant 响应，重试计数器立即归零。这意味着一个 turn 内的多次 LLM 调用（比如多个工具调用后续的 LLM 调用）各自独立计数，不会因为前一次的重试消耗后续调用的重试配额。

### 溢出恢复

当实际 token 超出估算，LLM 返回 context overflow 错误时，Pi 的恢复流程：

```
LLM 返回 context overflow 错误
        │
        ▼
检查 _overflowRecoveryAttempted
        │
    ┌───┴───┐
    │ false │          │ true │
    └───┬───┘          └──┬───┘
        │                 │
   设为 true          放弃恢复
   移除错误消息       提示用户切换
   触发 compaction    更大的模型
   压缩完成后重试
```

这个单次限制避免了无限循环 —— 如果压缩后还是溢出，说明问题不在历史消息太多，可能是 system prompt 本身太大或 `keepRecentTokens` 设置过高。

## 6.6 成本与约束

构建 Agent 时，成本是不可忽视的：

**Token 计费公式**：
```
cost = (input_tokens × price_per_M / 1,000,000) + (output_tokens × price_per_M / 1,000,000)
```

**一次 Agent Run 的典型成本**：

| 场景 | 轮数 | Token 消耗 | 费用（Claude Sonnet） |
|------|------|-----------|---------------------|
| 简单问答 | 1 轮 | ~2K | ~$0.01 |
| 修复一个 bug | 3-5 轮 | 10-30K | $0.03-0.10 |
| 大型重构 | 10-20 轮 | 50-200K | $0.15-0.60 |

**控制成本的手段**：
- `maxTurns`: 限制最大轮数（Demo 06 设了 20 轮上限）
- `maxTokens`: 限制每轮输出长度
- Compaction: 在 context 接近 window 限制时自动压缩
- 模型选择: 简单工具调用用便宜模型，复杂推理用贵模型

## 6.7 三种运行模式

Pi 支持三种运行模式，所有模式共享同一个 `AgentSession` 核心：

```
                  ┌─────────────┐
                  │ AgentSession │
                  │  (核心逻辑)  │
                  └──────┬──────┘
                         │
           ┌─────────────┼─────────────┐
           ▼             ▼             ▼
   ┌───────────┐  ┌───────────┐  ┌───────────┐
   │Interactive│  │   Print   │  │    RPC    │
   │  (TUI)   │  │  (单次)   │  │ (IDE集成) │
   └───────────┘  └───────────┘  └───────────┘
```

### Interactive 模式 (TUI)

这是 Pi 的默认模式。`InteractiveMode` 类实现了一个完整的终端 UI，包含编辑器、消息渲染、工具进度显示等。核心文件 `interactive-mode.ts` 有 **4,570 行**。

```typescript
// packages/coding-agent/src/modes/interactive/interactive-mode.ts
export class InteractiveMode {
  private session: AgentSession;
  private ui: TUI;
  // 处理键盘事件、渲染消息、管理编辑器状态...
  // Ctrl+P 切换模型、Ctrl+T 切换 thinking level
  // ESC-ESC 打开 tree view、/ 激活 slash commands
  async run(): Promise<void> { /* 主事件循环 */ }
}
```

### Print 模式

非交互式，适合 CI/CD 或脚本调用。执行一次 prompt，输出结果，退出。

```typescript
// 命令行使用:
// pi -p "Fix the bug in auth.ts"        ← text 模式
// pi -p "Fix the bug" --output-format json  ← JSON 事件流

export async function runPrintMode(
  session: AgentSession,
  options: PrintModeOptions,
): Promise<void> {
  // 发送消息 → 收集输出 → 写到 stdout → 退出
}
```

### RPC 模式

为 IDE 集成设计。通过 stdin/stdout 的 JSONL 协议通信：

```typescript
// 启动: pi --rpc
// IDE 通过 stdin 发送 JSON 命令，通过 stdout 接收事件

export async function runRpcMode(session: AgentSession): Promise<never> {
  // stdin: { type: "prompt", text: "fix bug" }
  // stdout: { type: "text_delta", text: "I'll..." }
  // stdout: { type: "tool_start", name: "read", ... }
  // ...
}
```

三种模式的共同点是：它们都不包含业务逻辑。所有的 prompt 处理、工具执行、重试、压缩都在 `AgentSession` 中完成。模式只负责 I/O 层面的适配。

## 6.9 毕业项目：设计你自己的 Agent

学了 6 章，是时候构建自己的 Agent 了 — 不是复制 Pi，不是复制 Demo 06，而是解决一个你自己的问题。

### 关键洞察：Agent Loop 是通用的

先看一个 50 行的"天气查询 Agent"：

```typescript
// 和 Ch01 的 30 行版本结构完全相同 — 只换了工具和 prompt
const tools = [{
  name: 'get_weather',
  description: '查询城市天气',
  input_schema: {
    type: 'object',
    properties: { city: { type: 'string', description: '城市名' } },
    required: ['city']
  }
}];

function executeTool(name: string, input: any): string {
  if (name === 'get_weather') {
    // 模拟天气 API
    const weather: Record<string, string> = {
      '北京': '晴，25°C，湿度 60%',
      '上海': '多云，22°C，湿度 75%',
    };
    return weather[input.city] || `未找到 ${input.city} 的天气数据`;
  }
  return 'Unknown tool';
}

// Agent Loop — 和 Ch01 完全一样！
const messages = [{ role: 'user', content: '北京和上海今天哪个城市更热？' }];

while (true) {
  const response = await client.messages.create({
    model: 'claude-sonnet-4-5-20250514',
    max_tokens: 1024,
    tools,
    system: '你是天气助手。用 get_weather 工具查询天气后回答用户。',
    messages,
  });

  messages.push({ role: 'assistant', content: response.content });
  if (response.stop_reason !== 'tool_use') break;

  const toolResults = [];
  for (const block of response.content) {
    if (block.type === 'tool_use') {
      const result = executeTool(block.name, block.input);
      toolResults.push({ type: 'tool_result', tool_use_id: block.id, content: result });
    }
  }
  messages.push({ role: 'user', content: toolResults });
}
```

**看到了吗？Agent Loop 完全不变，你只需要换工具和 prompt。这就是通用框架的力量。**

## 6.11 本章小结

本章是整个教程的 capstone。我们把前 5 章学到的概念整合成一个完整的 coding agent：

### 概念回顾

| 章节 | 核心概念 | 在 Demo 06 中的体现 |
|------|---------|-------------------|
| **第 1 章** Agent 全景 | LLM + Tools + Loop | `Agent` 类 = `Anthropic` + `TOOL_EXECUTORS` + `while` |
| **第 2 章** 核心循环 | 双层循环、stop_reason | `prompt()` 的 while 循环 + tool_use 检测 |
| **第 3 章** 工具系统 | Schema → 验证 → 执行 | `tools/` 目录：4 个工具各自定义 Schema + executor |
| **第 4 章** 流式处理 | EventStream、text_delta | `AgentEvent` 类型 + `onEvent` 回调 |
| **第 5 章** 多模型 | Provider 抽象、API Key 解析 | `config-resolver.ts` 的 provider/model 配置 |
| **第 6 章** 完整 Agent | 上面全部 + 会话/压缩/重试/配置 | 7 个文件的完整实现 |

### Pi 的设计哲学

通过 6 章的学习，我们可以提炼出 Pi 的核心设计原则：

**1. 分层不分裂** — `pi-ai` / `pi-agent-core` / `pi-coding-agent` 三层各有清晰职责，但通过统一的类型系统无缝连接。`AgentMessage` 从最底层的 `Message` 扩展，贯穿所有层。

**2. 纯函数 + 状态隔离** — `agentLoop()` 是纯函数，`buildSystemPrompt()` 是纯函数，`compact()` 是纯函数。状态被封装在 `AgentSession` 和 `SessionManager` 中，I/O 被推到运行模式层。

**3. 自适应而非硬编码** — system prompt 根据工具集自适应，guidelines 根据启用功能动态生成，compaction 根据 token 使用量自动触发。

**4. 安全边界无处不在** — `MAX_TURNS` 防止无限循环，`_overflowRecoveryAttempted` 防止无限重试，`isRetryable()` 区分可重试和不可重试错误，`chmod 0o600` 保护 API key 文件。

**5. 渐进式复杂度** — Demo 06 用 ~400 行代码实现了 Pi ~10,000 行代码的核心功能子集。Pi 的额外复杂度来自生产级需求：树形会话、版本迁移、多模式支持、扩展系统、TUI 渲染。但核心理念完全一致。

### 从 Demo 到生产

如果你想把 Demo 06 发展成一个生产级 Agent，以下是关键的升级路径：

```
Demo 06 (当前)                    Production Agent
─────────────────                 ─────────────────
线性 JSONL session         →      树结构 + fork/branch
chars/4 无 compaction      →      自动 compaction + 结构化摘要
简单重试                    →      重试 + 溢出恢复 + 成功重置
readline REPL              →      TUI / Print / RPC 三模式
4 个工具                    →      7+ 工具 + 扩展系统
单模型                      →      多模型 + 热切换
硬编码 guidelines          →      自适应 guidelines + Skills
```

每一步都是前一步的自然延伸，没有架构上的断裂。这就是为什么我们说 Pi 的设计是**渐进式**的 —— 从最简单的 while 循环开始，一步步叠加能力，最终成为一个功能完整的 coding agent。
