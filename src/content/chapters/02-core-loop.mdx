export const codeRunAgentLoop = `export async function runAgentLoop(
  prompts: AgentMessage[],
  context: AgentContext,
  config: AgentLoopConfig,
  emit: AgentEventSink,
  signal?: AbortSignal,
  streamFn?: StreamFn,
): Promise<AgentMessage[]> {
  const newMessages: AgentMessage[] = [...prompts];
  const currentContext: AgentContext = {
    ...context,
    messages: [...context.messages, ...prompts],
  };

  await emit({ type: "agent_start" });
  await emit({ type: "turn_start" });
  for (const prompt of prompts) {
    await emit({ type: "message_start", message: prompt });
    await emit({ type: "message_end", message: prompt });
  }

  await runLoop(currentContext, newMessages, config, signal, emit, streamFn);
  return newMessages;
}`;

export const codeRunLoop = `async function runLoop(
  currentContext: AgentContext,
  newMessages: AgentMessage[],
  config: AgentLoopConfig,
  signal: AbortSignal | undefined,
  emit: AgentEventSink,
  streamFn?: StreamFn,
): Promise<void> {
  let firstTurn = true;
  // Check for steering messages at start (user may have typed while waiting)
  let pendingMessages: AgentMessage[] = (await config.getSteeringMessages?.()) || [];

  // Outer loop: continues when queued follow-up messages arrive after agent would stop
  while (true) {
    let hasMoreToolCalls = true;

    // Inner loop: process tool calls and steering messages
    while (hasMoreToolCalls || pendingMessages.length > 0) {
      if (!firstTurn) {
        await emit({ type: "turn_start" });
      } else {
        firstTurn = false;
      }

      // Process pending messages (inject before next assistant response)
      if (pendingMessages.length > 0) {
        for (const message of pendingMessages) {
          await emit({ type: "message_start", message });
          await emit({ type: "message_end", message });
          currentContext.messages.push(message);
          newMessages.push(message);
        }
        pendingMessages = [];
      }

      // Stream assistant response
      const message = await streamAssistantResponse(currentContext, config, signal, emit, streamFn);
      newMessages.push(message);

      if (message.stopReason === "error" || message.stopReason === "aborted") {
        await emit({ type: "turn_end", message, toolResults: [] });
        await emit({ type: "agent_end", messages: newMessages });
        return;
      }

      // Check for tool calls
      const toolCalls = message.content.filter((c) => c.type === "toolCall");
      hasMoreToolCalls = toolCalls.length > 0;

      const toolResults: ToolResultMessage[] = [];
      if (hasMoreToolCalls) {
        toolResults.push(...(await executeToolCalls(currentContext, message, config, signal, emit)));

        for (const result of toolResults) {
          currentContext.messages.push(result);
          newMessages.push(result);
        }
      }

      await emit({ type: "turn_end", message, toolResults });

      pendingMessages = (await config.getSteeringMessages?.()) || [];
    }

    // Agent would stop here. Check for follow-up messages.
    const followUpMessages = (await config.getFollowUpMessages?.()) || [];
    if (followUpMessages.length > 0) {
      // Set as pending so inner loop processes them
      pendingMessages = followUpMessages;
      continue;
    }

    // No more messages, exit
    break;
  }

  await emit({ type: "agent_end", messages: newMessages });
}`;

export const codeStreamAssistantResponse = `async function streamAssistantResponse(
  context: AgentContext,
  config: AgentLoopConfig,
  signal: AbortSignal | undefined,
  emit: AgentEventSink,
  streamFn?: StreamFn,
): Promise<AssistantMessage> {
  // Apply context transform if configured (AgentMessage[] → AgentMessage[])
  let messages = context.messages;
  if (config.transformContext) {
    messages = await config.transformContext(messages, signal);
  }

  // Convert to LLM-compatible messages (AgentMessage[] → Message[])
  const llmMessages = await config.convertToLlm(messages);

  // Build LLM context
  const llmContext: Context = {
    systemPrompt: context.systemPrompt,
    messages: llmMessages,
    tools: context.tools,
  };

  const streamFunction = streamFn || streamSimple;
  // ... resolve API key, call LLM ...
  const response = await streamFunction(config.model, llmContext, {
    ...config,
    apiKey: resolvedApiKey,
    signal,
  });

  let partialMessage: AssistantMessage | null = null;
  let addedPartial = false;

  for await (const event of response) {
    switch (event.type) {
      case "start":
        partialMessage = event.partial;
        context.messages.push(partialMessage);
        addedPartial = true;
        await emit({ type: "message_start", message: { ...partialMessage } });
        break;

      case "text_start":
      case "text_delta":
      // ... other streaming events ...
        if (partialMessage) {
          partialMessage = event.partial;
          context.messages[context.messages.length - 1] = partialMessage;
          await emit({
            type: "message_update",
            assistantMessageEvent: event,
            message: { ...partialMessage },
          });
        }
        break;

      case "done":
      case "error": {
        const finalMessage = await response.result();
        if (addedPartial) {
          context.messages[context.messages.length - 1] = finalMessage;
        } else {
          context.messages.push(finalMessage);
        }
        await emit({ type: "message_end", message: finalMessage });
        return finalMessage;
      }
    }
  }
  // ...
}`;

export const codeProcessLoopEvent = `private _processLoopEvent(event: AgentEvent): void {
  switch (event.type) {
    case "message_start":
      this._state.streamMessage = event.message;
      break;

    case "message_update":
      this._state.streamMessage = event.message;
      break;

    case "message_end":
      this._state.streamMessage = null;
      this.appendMessage(event.message);
      break;

    case "tool_execution_start": {
      const pendingToolCalls = new Set(this._state.pendingToolCalls);
      pendingToolCalls.add(event.toolCallId);
      this._state.pendingToolCalls = pendingToolCalls;
      break;
    }

    case "tool_execution_end": {
      const pendingToolCalls = new Set(this._state.pendingToolCalls);
      pendingToolCalls.delete(event.toolCallId);
      this._state.pendingToolCalls = pendingToolCalls;
      break;
    }

    case "turn_end":
      if (event.message.role === "assistant" && (event.message as any).errorMessage) {
        this._state.error = (event.message as any).errorMessage;
      }
      break;

    case "agent_end":
      this._state.isStreaming = false;
      this._state.streamMessage = null;
      break;
  }

  this.emit(event);
}`;

steer(m: AgentMessage) {
  this.steeringQueue.push(m);

/** Queue a follow-up message to be processed after the agent finishes. */
followUp(m: AgentMessage) {
  this.followUpQueue.push(m);

private dequeueSteeringMessages(): AgentMessage[] {
  if (this.steeringMode === "one-at-a-time") {
    if (this.steeringQueue.length > 0) {
      const first = this.steeringQueue[0];
      this.steeringQueue = this.steeringQueue.slice(1);
      return [first];
    }
    return [];
  }

  const steering = this.steeringQueue.slice();
  this.steeringQueue = [];
  return steering;
}

      return [first];
    }
    return [];

  const followUp = this.followUpQueue.slice();
  this.followUpQueue = [];
  return followUp;
}`;

// This is the core pattern: while(true), call LLM, check if done, execute tools, repeat
while (true) {
  turnCount++;
  printEvent(\`Turn \${turnCount}\`, "调用 LLM...");

  );

  if (response.stop_reason === "end_turn" || toolUseBlocks.length === 0) {
    printEvent("Done", \`Agent 完成 (stop_reason=\${response.stop_reason})\`);
    break;

      tool_use_id: toolUse.id,
      content: result,
    });
  }

  // Add assistant response and tool results to message history
  messages.push({ role: "assistant", content: response.content });
  messages.push({ role: "user", content: toolResults });
}`;

  steeringQueue.push(message);
  printEvent("STEERING_ADDED", \`"\${message}"\`);

export const codeAbortHandler = `async function runAgentWithAbort(
  client: Anthropic,
  task: string,
  signal: AbortSignal,
): Promise<{ completed: boolean; turns: number }> {
  const messages: Anthropic.MessageParam[] = [
    { role: "user", content: task },
  ];
  let turnCount = 0;
  const MAX_TURNS = 20;

  while (turnCount < MAX_TURNS) {
    // Check abort before LLM call
    if (signal.aborted) {
      printEvent("ABORT", "Signal detected before LLM call — exiting loop");
      return { completed: false, turns: turnCount };
    }

    turnCount++;
    let response: Anthropic.Message;
    try {
      response = await client.messages.create({
        model: "claude-sonnet-4-5-20250514",
        max_tokens: 1024,
        tools,
        messages,
      });
    } catch (err) {
      if (signal.aborted) {
        printEvent("ABORT", "LLM call interrupted by abort signal");
        return { completed: false, turns: turnCount };
      }
      throw err;
    }

    // Check abort after LLM call
    if (signal.aborted) {
      printEvent("ABORT", "Signal detected after LLM call — exiting loop");
      return { completed: false, turns: turnCount };
    }

    // ... 正常处理 ...

    for (const toolUse of toolUseBlocks) {
      // Check abort before each tool execution
      if (signal.aborted) {
        printEvent("ABORT", \`Signal detected before tool \${toolUse.name} — exiting\`);
        return { completed: false, turns: turnCount };
      }

      const result = await executeTool(toolUse.name, toolUse.input);
      // ...
    }
  }
}`;

# 核心循环

本章目标：深入 Pi 的核心 — `agent-loop.ts`，理解双层循环的设计，掌握 Steering 和 Follow-up 机制，最终用 6 个递进的 demo 从零实现一个 mini agent loop。

这是整个教程最重要的一章。上一章我们看到"Agent 的本质是一个 while 循环"，但从 30 行 demo 到生产级框架，中间还有巨大的鸿沟。Pi 用约 616 行代码把这个鸿沟填满，而且填得非常优雅。

### 从 100 行到 616 行：差了什么？

上一章的 100 行 Agent 已经有了多工具、事件和错误处理。但如果你用它做一个真实任务——比如让它把项目里 12 个文件的旧 API 调用迁移到新 SDK——你会立刻遇到三个问题：

1. **你无法在 Agent 执行到一半时插话。** Agent 正在分析第 5 个文件，你突然想说"跳过测试文件"，但没有任何入口让你把这条指令送进正在运行的循环。你只能杀掉进程重来。

2. **Agent 完成后你无法追问。** Agent 改完所有文件停下来了，你想说"再检查一下有没有遗漏"。但 while 循环已经 break 了，你只能重新启动一个新的 Agent 实例。

3. **你无法优雅地中止。** 按 Ctrl+C 是进程级别的暴力终止，正在写入的文件可能只写了一半。

这三个问题对应 Pi 的三个核心机制：**Steering**（中途插话）、**Follow-up**（追问续命）、**Abort**（优雅中断）。它们是 100 行到 616 行之间最核心的 500 行。

### 从 30 行到 600 行：三个问题

在看 `runLoop` 源码之前，先想想你在 Chapter 1 写的 30 行 Agent 会遇到什么问题：

**问题 1：用户想中途插话。** Agent 正在执行第 3 轮工具调用，用户突然说"等等，先别删那个文件"。你的 while 循环没有检查点来接收这个消息。

**问题 2：Agent 完成了，但用户想追问。** Agent 回答完毕退出了循环，但用户想说"再帮我检查一下测试"。你得重新调用 `prompt()`，但上下文已经丢了。

**问题 3：出错了怎么办？** LLM 返回了格式错误的响应，或者工具执行超时了。你的 while 循环会崩溃。

这就是 Pi 的 `runLoop` 要解决的三个核心问题。它的答案是：**双层循环 + Steering/Follow-up 队列**。

先定义两个关键概念：

- **Steering（中断注入）**：在工具执行间隙，把用户的新消息注入到下一轮 LLM 调用前。可以理解为 "interrupt" — Agent 还在跑，但你需要它听你一句新指令。
- **Follow-up（续命）**：Agent 想停的时候，检查有没有追加消息，有的话继续执行。可以理解为 "keep going" — Agent 觉得自己做完了，但你还有新任务给它。

带着这两个概念，来读 `runLoop` 的源码。

### Steering vs Follow-up：两种"插话"机制

前面我们定义了 Steering（中断注入）和 Follow-up（续命）。现在对照源码，完整对比两种机制：

| | Steering（中断） | Follow-up（续命） |
|---|---|---|
| **时机** | 内层循环每轮结束后 | 外层循环，Agent 准备停止时 |
| **语义** | "等等，我还有补充" | "你做完了？我还有新任务" |
| **效果** | 在下一次 LLM 调用前注入消息 | 把 Agent 从"准备停止"拉回来继续运行 |
| **典型场景** | 用户在 Agent 执行工具时输入新指令 | Agent 完成一个任务后，用户追加新任务 |
| **Pi 默认模式** | one-at-a-time | one-at-a-time |

## 2.3 边界行为与陷阱

理解 Pi 的设计不仅要看正常流程，更要看边界情况。这些细节决定了框架的健壮性。

### Steering one-at-a-time vs all

```
场景：用户在 Agent 执行工具期间快速输入了 3 条 steering 消息

one-at-a-time 模式（Pi 默认）:
  Turn 5: [工具执行完毕] → 取出 steering[0] → 注入 → LLM 回复
  Turn 6: [工具执行完毕] → 取出 steering[1] → 注入 → LLM 回复
  Turn 7: [工具执行完毕] → 取出 steering[2] → 注入 → LLM 回复

all 模式:
  Turn 5: [工具执行完毕] → 取出全部 3 条 → 一次性注入 → LLM 回复（需要同时理解 3 条指令）
```

**one-at-a-time 的优势**：LLM 每次只处理一条新指令，认知负担更小。如果第一条 steering 就改变了方向，后面的 steering 仍然能基于新方向的上下文执行。

**all 的优势**：减少 LLM 调用次数。如果 3 条 steering 是相关的（比如"用中文"+"用表格"+"加总结"），一次性给出效果更好。

### Follow-up 只在"真的要停"时触发

Follow-up 的检查时机非常精确 — 只在内层循环完全退出后才检查：

```
Agent 工作中...
  内层循环检查：有工具调用？  → 是 → 继续内层循环
  内层循环检查：有 steering？ → 是 → 继续内层循环
  内层循环检查：有工具调用？  → 否
  内层循环检查：有 steering？ → 否
  → 内层循环退出

外层循环检查：有 follow-up？ → 是 → 把 follow-up 变成 pendingMessages → 重新进入内层循环
外层循环检查：有 follow-up？ → 否 → 真正结束
```

关键区别：**Steering 在工具执行的间隙就会被消费，Follow-up 要等到 Agent 认为自己"完成了"才会被消费。**

### abort() 在不同阶段的行为

```
阶段 1: LLM 正在 streaming
  → abort() 发出信号
  → streamSimple 收到 AbortSignal，停止接收
  → 返回 stopReason="aborted" 的 AssistantMessage
  → runLoop 检查到 aborted，立即 return

阶段 2: 工具正在执行（sequential 模式）
  → abort() 发出信号
  → 当前工具执行完毕（工具自己决定是否 honor signal）
  → 下一个工具执行前检查 signal → 跳过
  → 回到 runLoop，在下一次 LLM 调用前退出

阶段 3: 工具并发执行（parallel 模式）
  → abort() 发出信号
  → 所有正在执行的工具收到 signal
  → 各工具自行决定是否提前退出
  → 全部完成后回到 runLoop，退出
```

重点：**abort 是协作式的**。Pi 不会强制 kill 正在执行的工具，而是把 AbortSignal 传递给工具，让工具自己决定如何处理。

**AbortSignal 传播链**：从 `Agent.abort()` 开始，signal 沿着整个调用链逐层传递：

```
Agent.abort()
  └→ AbortController.abort()
       ├→ streamFn(signal) — LLM 流式调用检查 signal
       ├→ beforeToolCall(ctx, signal) — 钩子检查 signal
       ├→ tool.execute(id, args, signal) — 工具检查 signal
       └→ afterToolCall(ctx, signal) — 钩子检查 signal
```

每一层都接收同一个 signal，任何一层都可以在检测到 `signal.aborted` 时提前退出。这就是 Web 标准 AbortController 的设计哲学 — 用一个 signal 串联所有异步操作，实现统一的取消语义。

### pendingToolCalls 每次创建新 Set

```typescript
case "tool_execution_start": {
  const pendingToolCalls = new Set(this._state.pendingToolCalls);
  pendingToolCalls.add(event.toolCallId);
  this._state.pendingToolCalls = pendingToolCalls;  // ← 新引用！
  break;
}
```

为什么不直接 `this._state.pendingToolCalls.add(event.toolCallId)` ？因为 React/Solid/Vue 等 UI 框架靠 **引用比较** 判断是否需要重渲染。如果只是修改 Set 内容但引用不变，UI 不会更新。每次创建新 Set 保证引用变化，从而触发 UI 响应。

### replaceMessages 的浅拷贝 vs appendMessage 的 spread

```typescript
replaceMessages(ms: AgentMessage[]) {
  this._state.messages = ms.slice();    // ← 浅拷贝：切断与调用者的引用关系
}

appendMessage(m: AgentMessage) {
  this._state.messages = [...this._state.messages, m];  // ← 创建新数组：触发响应式更新
}
```

两个方法的模式一致：**每次都产生新的数组引用**。`replaceMessages` 用 `slice()` 断开与外部数组的引用，`appendMessage` 用 spread 创建新数组。如果直接 `push(m)`，外部的响应式系统不会感知到变化。

### State mutators 不 emit 事件

```typescript
setSystemPrompt(v: string) {
  this._state.systemPrompt = v;
  // 注意：没有 emit 任何事件！
}

setModel(m: Model<any>) {
  this._state.model = m;
  // 同样没有 emit
}
```

这是一个有意的设计决策：**配置变更不触发事件**。事件只在 Agent 运行期间（`_processLoopEvent` 中）由 loop 事件驱动。如果 UI 需要响应配置变更，应该直接读 `agent.state`，而不是监听事件。

### Demo 06: 边界情况

真实 Agent 框架需要处理的边界情况远比核心循环复杂。Demo 06 测试了三个关键场景：

**场景 1：工具错误恢复**

```typescript
// 工具抛出错误时，不是让整个 Agent 崩溃
// 而是把错误作为 tool_result 返回给 LLM
function safeExecuteTool(name: string, input: Record<string, unknown>)
  : { result: string; isError: boolean } {
  try {
    const result = executeTool(name, input);
    return { result, isError: false };
  } catch (err) {
    return { result: `Error: ${(err as Error).message}`, isError: true };
  }
}

// 把错误发回 LLM 时标记 is_error: true
toolResults.push({
  type: "tool_result",
  tool_use_id: toolUse.id,
  content: result,
  is_error: isError,  // ← 告诉 LLM 这是错误，不是正常结果
});
```

这对应 Pi 的 `createErrorToolResult` 函数和 `prepareToolCall` 中的 try-catch 逻辑。LLM 收到错误后通常会自动调整策略 — 比如除以零失败后改用其他除数。

**场景 2：Steering 队列模式对比**

```
one-at-a-time 模式（Pi 默认）:
  队列: ["请用中文回答", "格式用表格", "加上总结"]
  本轮注入: "请用中文回答"
  队列剩余: ["格式用表格", "加上总结"]
  → 每轮消耗一条

all 模式:
  队列: ["请用中文回答", "格式用表格", "加上总结"]
  本轮注入: ["请用中文回答", "格式用表格", "加上总结"]
  队列剩余: []
  → 一次性全部消耗
```

**场景 3：空内容处理**

LLM 偶尔会返回空的 `content` 数组。Pi 的 `runLoop` 通过检查 `toolCalls.length > 0` 和 `hasMoreToolCalls` 来处理这种情况 — 没有工具调用时直接退出内层循环。

## 2.5 本章小结

四个核心认知：

1. **Agent Loop 本质是双层 while 循环** — 内层处理 tool calls + steering，外层处理 follow-up。这个结构看似简单，但精确的检查时机和状态管理让它能覆盖所有运行时场景。

2. **Steering = 中断（工具间隙注入），Follow-up = 续命（Agent 要停时拉回来）** — 两种机制的触发时机不同、语义不同，但都通过消息队列实现，且都支持 one-at-a-time 和 all 两种消费模式。

3. **事件驱动是 Agent 架构的关键设计** — Pi 定义了 12 种 AgentEvent，覆盖 Agent > Turn > Message/Tool 三层嵌套的完整生命周期。事件是 async emit 的，保证严格有序。Agent 类先更新自身状态，再广播给外部监听者。

4. **生产级框架的差距在边界处理** — 从 30 行 demo 到 Pi 的 1,427 行核心代码，增加的主要是：流式输出、两层转换管道、工具三阶段生命周期（prepare → execute → finalize）、响应式状态管理、以及大量的防御性错误处理。

下一章，我们将深入 Pi 的工具系统 — 看看 `AgentTool` 的 TypeBox schema 验证、`beforeToolCall`/`afterToolCall` hooks、以及 coding agent 的 7 个内置工具是如何实现的。
