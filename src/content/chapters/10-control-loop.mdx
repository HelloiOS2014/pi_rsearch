export const codeEventQueue = `// 事件队列 — 用 Promise 链保证串行处理
private _agentEventQueue: Promise<void> = Promise.resolve();

private _handleAgentEvent = (event: AgentEvent): void => {
  // 同步创建 retry promise（避免竞态）
  this._createRetryPromiseForAgentEnd(event);

  // 将事件处理串入 Promise 链，保证顺序执行
  this._agentEventQueue = this._agentEventQueue.then(
    () => this._processAgentEvent(event),
    () => this._processAgentEvent(event), // 前一个失败也继续处理
  );

  // 防止未捕获的 rejection 导致队列断裂
  this._agentEventQueue.catch(() => {});
};`;

export const codeAutoRetry = `// 自动重试 — 指数退避，可中断，错误分类
private _isRetryableError(message: AssistantMessage): boolean {
  if (message.stopReason !== "error") return false;
  // Context Overflow 由压缩处理，不走重试
  if (isContextOverflow(message, contextWindow)) return false;
  // 匹配可重试的错误模式：限流、过载、网络错误
  return /overloaded|rate.?limit|429|500|502|503|504|
    service.?unavailable|network.?error|connection.?error|
    fetch failed|timed? out|timeout/i.test(message.errorMessage);
}

private async _handleRetryableError(message: AssistantMessage) {
  this._retryAttempt++;
  if (this._retryAttempt > settings.maxRetries) {
    this._emit({ type: "auto_retry_end", success: false });
    return false;  // 超出上限，放弃
  }
  // 指数退避：1s → 2s → 4s → 8s...
  const delayMs = settings.baseDelayMs * 2 ** (this._retryAttempt - 1);
  await sleep(delayMs, this._retryAbortController.signal);
  // 从 Agent 状态中移除错误消息，重新发送
  this.agent.replaceMessages(messages.slice(0, -1));
  setTimeout(() => this.agent.continue(), 0);
  return true;
}`;

export const codeOverflowRecovery = `// Context Overflow 恢复 — 压缩 + 重试，仅尝试一次
private _overflowRecoveryAttempted = false;

private async _checkCompaction(assistantMessage: AssistantMessage) {
  // Case 1: Overflow — LLM 报告上下文溢出
  if (isContextOverflow(assistantMessage, contextWindow)) {
    if (this._overflowRecoveryAttempted) {
      // 已经试过一次，不再重复——避免无限循环
      this._emit({
        type: "auto_compaction_end",
        errorMessage: "Context overflow recovery failed after one attempt.",
      });
      return;
    }
    this._overflowRecoveryAttempted = true;
    // 从 Agent 状态中移除错误消息
    this.agent.replaceMessages(messages.slice(0, -1));
    // 触发自动压缩，压缩完成后自动重试
    await this._runAutoCompaction("overflow", true);
    return;
  }

  // Case 2: Threshold — 上下文接近上限，预防性压缩
  if (shouldCompact(contextTokens, contextWindow, settings)) {
    await this._runAutoCompaction("threshold", false);
  }
}`;

## Agent 失控了

> 你在 Demo 06 中构建的 Agent 已经能跑起来了。它读文件、写代码、调工具，终端里的输出流畅而自信。你觉得是时候让它干点正事了——把项目里散落的旧 API 调用迁移到新版 SDK。
>
> **第 5 分钟**，你发现 Agent 走偏了。它开始修改一个不需要改的文件，而且越改越深入。你想纠正它——但没有任何办法。Agent 正在执行一个 LLM 调用，你发出的消息进不去，只能等它把当前 turn 跑完。你盯着屏幕看它把一个无辜的 `utils.ts` 改得面目全非，束手无策。你唯一的选项是 Ctrl+C 杀掉进程。
>
> **第 10 分钟**，你重启了 Agent，重新给了指令。它干了三分钟，然后 API 返回 `429 Too Many Requests`。Agent 把这个错误当成了任务的一部分，尝试"理解"它，输出了一段关于 HTTP 状态码的解释，然后停了下来。没有重试，没有退避，没有恢复——一个临时的限流错误，直接终结了整个工作流。
>
> 你再次重启。Agent 像第一次见到你一样：`"Hello! How can I help you today?"` 它不记得已经迁移了 4 个文件。不记得你说过 `auth.ts` 要特殊处理。不记得你们之间关于 error boundary 的讨论。一切从零开始。
>
> 第三次尝试。这回走得远了一些，但 Agent 在编辑 `config.ts` 时遇到了一个字符匹配失败。它没有换个策略，而是用完全相同的方式重试了 5 次——相同的 diff，相同的错误，相同的结果。循环，循环，循环。
>
> 到了第四次尝试，你学聪明了，在 Agent 工作时输入了一条补充指令："对了，`config.ts` 里的引号是中文引号，注意处理。" 你等了两分钟，Agent 完全忽略了你的消息。后来你才发现，这条消息被排进了一个队列，要等当前 turn 结束才会投递——但当前 turn 是一个长达 8 分钟的多步骤编辑链。
>
> 你回头看了看代码。Agent 的核心循环完好无损——`while` 循环在转，LLM 在推理，Tool 在执行。一切都在"正常工作"。
>
> 但它无法被中途修正。它无法从瞬时错误中恢复。它在崩溃后失去全部记忆。它在失败时无限重复。它听不到你的实时反馈。

你的 Agent 有一个大脑，但没有神经系统。

大脑负责思考——LLM 推理、Tool 调用、代码生成。这些你在 Part 1 中已经构建好了。但一个只有大脑的系统是脆弱的。它缺少的是将大脑连接到现实世界的控制回路：感知环境变化、处理异常信号、维持运行状态、接收外部指令。

这正是从 Demo 到产品之间那条看不见的鸿沟。

## 为什么会痛：控制系统视角

回顾 Ch9 的控制系统模型，Agent Harness 的核心职责之一就是实现对 Agent 行为的闭环控制。上面的每一个痛点，都对应着控制理论中一个具体的失败模式。

**无法中途修正 → 可控性不足（Controllability）。** 一个可控的系统，意味着操作者可以在任意时刻将系统从当前状态引导到期望状态。你的 Agent 在 turn 执行期间拒绝接受外部输入——它的可控性在每个 turn 内降到了零。这不是"功能缺失"，是控制论意义上的系统不可控。

**API 限流导致崩溃 → 扰动抑制缺失（Disturbance Rejection）。** 在控制理论中，外部干扰（噪声、扰动）是系统必须处理的常态，不是异常。429 错误是一个典型的可预测扰动——它的到来不是问题，系统没有扰动抑制回路才是问题。一个健壮的控制系统会检测扰动类型、选择恢复策略、自动重试。

**重启后失忆 → 状态不持续（State Persistence）。** 控制系统的状态是决策的基础。温度控制器记住当前温度，PID 控制器积累历史误差。当你的 Agent 崩溃重启后状态归零，它变成了一个无记忆的开环系统——每次行动都基于零信息，无法从历史中学习。

**消息投递延迟 → 控制带宽不足（Control Bandwidth）。** 控制带宽决定了系统对输入变化的响应速度。你的补充指令需要等 8 分钟才能到达 Agent——这意味着控制带宽远低于实际需求。在一个变化速度快于控制响应的系统中，操作者实际上丧失了控制权。

**相同失败重复 5 次 → 稳定性失败（Stability）。** 一个稳定的系统，当受到扰动后应该收敛回正常状态，而不是在错误模式中震荡。Agent 用相同的方式反复尝试相同的失败操作，这是教科书级的不稳定行为——正反馈循环导致系统锁死在错误状态。

这五个问题不是五个独立的 bug。它们是同一个架构缺失的五种症状：**你的 Agent 没有控制回路。**

Pi 如何解决这些问题？答案就在 `AgentSession` 这个类里——它是 Agent 的神经系统，将感知、决策和执行编织成一个完整的控制回路。接下来，我们一层一层地拆开它。

## 业界怎么解决

### Agent Orchestrator 模式

如果你有 Web 开发背景，理解 Orchestrator 最快的方式是类比应用服务器：

- **AgentLoop** ≈ 你写的应用代码（业务逻辑、路由处理）
- **Orchestrator** ≈ Tomcat / Express / Kestrel（生命周期、线程管理、中间件管道）

应用代码负责"做什么"，应用服务器负责"怎么活着"。Agent Orchestrator 的角色完全一致——它不参与推理，但它确保推理能可靠地发生。

一个完整的 Orchestrator 承担七项核心职责：

1. **生命周期管理** — 初始化、运行、暂停、恢复、优雅关闭
2. **事件分发** — Agent 事件路由到持久化、UI、扩展等消费者
3. **上下文组装** — system prompt + 历史消息 + 注入内容 → LLM 输入
4. **请求协调** — steering / follow-up / custom context 消息队列
5. **控制与守卫** — 停止条件、循环检测、权限评估、预算管控
6. **错误恢复** — 重试策略、溢出恢复、优雅降级
7. **可观测性** — 事件流设计、诊断、统计

不是每个产品都实现全部七项。它们在这条光谱上的位置，反映了截然不同的设计哲学。

### 设计光谱：从极薄到极厚

| 产品 | Orchestrator 厚度 | 为什么这么选 |
|------|------------------|-------------|
| **Vercel AI SDK** | 极薄 | SDK 哲学——提供原语，用户自己组装 |
| **Pi** | 中等（~3,180 行 AgentSession） | 极简哲学——够用就好，"The One Rule: 理解每一行" |
| **Claude Code** | 重量级 | 企业需求——18 种 Hook 事件、5 种权限模式、管理策略 |
| **LangGraph** | 图引擎 | 学术背景——用有向图形式化保证执行正确性 |

没有"正确"的厚度。Vercel AI SDK 薄是因为它是库——它的用户是开发者，开发者需要灵活性而非约束。Claude Code 厚是因为它直面企业环境——权限、审计、合规不是可选项。Pi 选择中间位置，是因为它的首要约束是可理解性：一个人能读懂全部代码，才能真正掌控系统。LangGraph 走了完全不同的路线，用图论的形式化保证取代了手工编排的脆弱性。你选哪种厚度，取决于你的约束和哲学，不取决于技术优劣。

### 自治光谱

Orchestrator 的设计还决定了 Agent 在自治光谱上的位置：

- **human-in-the-loop（每步审批）** — 最安全但最慢。每个工具调用都需要用户确认。适合高风险操作（生产部署、财务交易）。
- **human-on-the-loop（选择性干预）** — 平衡点。Agent 自主运行，但用户可以在任意时刻介入纠正。大多数编码 Agent（Pi、Claude Code、Cursor）在这里。
- **human-out-of-the-loop（仅监控）** — 最快但风险最高。Agent 完全自主，用户只看最终结果。适合低风险、高重复性任务。

Orchestrator 的权限系统、消息队列、停止条件共同决定了 Agent 落在光谱的哪个位置。同一个 Agent 核心，搭配不同的 Orchestrator 配置，可以在这三个级别之间自由切换——这正是 Pi 的 `AgentSession` 支持 Interactive / Print / RPC 三种模式的底层逻辑。

## Pi 的解法：AgentSession 解剖

Pi 的 `AgentSession` 大约 3,180 行——是整个 Harness 中最大的单一文件。它不是"Agent 的一部分"，它**就是** Orchestrator。Agent 负责推理和工具调用，AgentSession 负责一切其他：生命周期、事件路由、错误恢复、消息协调、上下文管理。接下来我们逐层拆解它的关键设计决策。

### 依赖注入与子系统协调

AgentSession 的第一个设计决策是：**所有协作者都通过构造函数的配置对象注入，内部不创建任何依赖。**

看 `AgentSessionConfig` 接口：`agent`、`sessionManager`、`settingsManager`、`resourceLoader`、`modelRegistry`——每一个都是外部创建、外部拥有、通过配置传入的。AgentSession 从不 `new` 任何一个协作者。它甚至不知道这些协作者的具体实现类型，只依赖接口。

构造函数的最后一行揭示了控制回路的起点：`this.agent.subscribe(this._handleAgentEvent)`。从这一刻起，Agent 产生的每一个事件——开始推理、工具调用、输出 token、遇到错误——都会流入 AgentSession 的事件处理管道。

**Theory anchor:** 为什么不在内部 `new`？控制系统原理——controller 不应该和 plant 耦合。同一个 AgentSession 驱动 Interactive、Print、RPC 三种模式（Ch14），因为它不依赖任何特定的交付界面。依赖注入让 Orchestrator 变成了一个纯粹的协调器：它知道"何时做"和"做什么"，但不知道"谁来做"——具体实现由外部决定。

每个事件通过 `_processAgentEvent` 后被四路分发：

- **SessionManager.append()** — 持久化到存储，确保崩溃恢复
- **ExtensionRunner.dispatch()** — 通知扩展钩子（权限检查、自定义逻辑）
- **UI callback** — 渲染到终端或 Web 界面
- **Auto-compaction check** — 检查上下文是否需要压缩

四路消费者都在同一个 Promise 链中依次执行——不是并行的。这看起来似乎低效，但这是刻意的设计选择。

**Theory anchor:** 为什么事件必须串行处理？控制论的基本原则——并发反馈会腐蚀控制器状态。如果持久化和 UI 渲染并发处理同一事件，可能出现持久化成功但 UI 未更新，或反过来。更危险的是，如果两个事件并发触发 auto-compaction 检查，可能导致重复压缩或状态不一致。Promise 链确保每个事件被完整处理后才处理下一个——这是用吞吐量换正确性的经典权衡。

### 消息队列与请求协调

当 Agent 正在运行时，外部（用户、扩展、系统）可能需要向它注入消息。但不同消息有不同的时序语义——有些需要立即中断当前 turn，有些要等 turn 结束，有些要等到下一次用户输入。AgentSession 用三条独立的队列处理这三种语义：

为什么不能合并成一条队列？因为它们的投递时机完全不同：

- **Steering 消息**是即时的。用户在 Agent 流式输出时敲入修正指令——"别改那个文件"——这条消息必须在当前 turn 的下一个 LLM 调用中出现，否则纠正就来不及了。
- **Follow-up 消息**是延迟的。Agent 完成一个 turn 后，系统可能需要追加指令——"现在运行测试"。这些消息排队等待，turn 结束后自动触发新 turn。
- **Next-turn 消息**是隐式的。扩展通过 `_pendingNextTurnMessages` 注入上下文信息（如文件变更通知），这些信息不触发新 turn，而是作为 context 附着在下一次用户输入上。

还有一个特殊队列 `_pendingBashMessages`——bash 工具的输出如果在 Agent 流式响应期间到达，会被暂存而不是立即注入，避免打断正在生成的回复。这是一个微妙但重要的细节：控制带宽（Ch9 提到的概念）不是越大越好，投递时机和顺序同样关键。

### 控制与守卫

Agent 什么时候该停下来？Pi 定义了五种停止条件：

1. **自然完成** — model 产出纯文本（无 tool call），表示它认为任务已完成
2. **Turn 限制** — 可配置的 `maxTurns`，超出后强制终止并报告
3. **预算限制** — `max_budget_usd` 跟踪累计花费，超出即停
4. **Abort 信号** — `AbortController` 由用户或系统触发，可在任意时刻中止
5. **不可恢复错误** — authentication failure、invalid model 等无法通过重试修复的错误

除了停止条件，Pi 还通过 tool hooks 实现了运行时守卫。`beforeToolCall` 在每次工具执行前触发——扩展可以检查工具名、参数，决定允许、拒绝或修改。`afterToolCall` 在工具返回后触发——可以审计、记录或变换结果。这套机制让权限控制完全外置于 Agent 核心，不同的部署环境可以注入不同的策略。

还有一个容易忽略的细节：Thinking Level 管理。当用户在会话中途切换模型（比如从 Sonnet 切到 Opus），thinking budget 需要相应调整。Pi 会自动 clamp thinking level 到目标模型支持的范围内——防止因配置不匹配导致 API 报错。

当 LLM 报告上下文溢出时，恢复流程是：检测溢出 → 移除错误消息 → 触发自动压缩（将历史对话摘要化）→ 压缩完成后自动重试。`_overflowRecoveryAttempted` 标志确保这个流程只执行一次——如果压缩后仍然溢出，说明问题不是"历史太长"而是"单条消息太大"，此时放弃比无限循环更明智。

**Theory anchor:** 这是成本反转原理的直接应用——溢出后"保留"所有历史已不可能，唯一出路是"压缩取回"。防循环标志是稳定性保证——防止"压缩 → 重试 → 再溢出 → 再压缩"的无限循环。

还有一层容易忽略的优雅降级：Extension 执行失败不会阻塞 Agent 主流程，Bash 命令执行失败作为 tool result 返回（而非抛异常）。这些设计确保了**隔离性**——一个子系统的故障不会级联到整个系统。

### 启动序列与优雅关闭

Pi 的启动不是简单的"创建实例然后跑"。从 `main.ts` 开始，经历六个阶段：

1. **CLI 参数解析** → 确定运行意图（interactive? print? resume?）
2. **Settings 加载** → 合并全局配置 + 项目级配置（`~/.pi/settings.json` + `.pi/settings.json`）
3. **Model Registry 构建** → 发现可用模型、验证 API Key、确定默认模型
4. **Resource Loading** → 加载 Skills、Prompts、Themes——这些都是运行时资源，不是硬编码
5. **Session 恢复或创建** → 如果有 `--resume` 标志，从持久化存储恢复上一次会话
6. **Mode 选择** → 根据运行意图创建 InteractiveMode / PrintMode / RpcMode

每个阶段都有明确的前置条件——Settings 加载失败会阻止后续阶段，Model Registry 构建失败会报告可用模型列表让用户选择。这不是"全部 try-catch"的粗暴方式，而是每个阶段独立验证、独立报错。

优雅关闭同样精心设计。Pi 维护了多个独立的 `AbortController`：compaction、retry、bash execution、branch summary 各有各的。当用户按 Ctrl+C 时，这些 controller 按优先级依次取消——先取消正在进行的 bash 命令，再取消压缩，最后取消 LLM 调用。不是一个大锤全部砸掉，而是精确控制每个子系统的关闭顺序。

### 可观测性

控制系统的"传感器"——如果你看不到系统内部发生了什么，你就无法控制它。

Pi 的可观测性建立在三个 API 上：

- **`subscribe(listener)`** — 一个 API，所有消费者。UI、持久化、扩展、诊断工具都通过同一个事件流获取信息。没有特殊通道，没有 VIP 访问。
- **`getSessionStats()`** — 返回结构化统计：消息数、工具调用数、token 总量、累计花费。
- **`getContextUsage()`** — 返回当前上下文使用百分比。接近 100% 时 UI 可以显示警告。

一个关键的设计决策：工具输出在事件流中是**完整的**，但发给 LLM 时是**截断的**。Host（UI、日志系统、扩展）永远拥有完整信息，Agent 只看到截断后的版本。这意味着当你需要诊断问题时——工具返回了什么？Agent 看到的是什么？——事件流里都有。

**Theory anchor:** 控制论中，可观测性定义为"能否从输出推断系统内部状态"。如果你只能看到 Agent 的最终输出而看不到中间事件流，系统对你是不可观测的——你无法诊断任何问题。Pi 的事件流让每一个内部状态转换都对外可见，这是可观测性的工程实现。

## 不同产品为什么做了不同选择

我们已经深入拆解了 Pi 的控制回路设计。但 Pi 只是众多方案中的一种。不同产品面对相同的控制问题，做出了截然不同的选择：

| 维度 | Pi | Claude Code | LangGraph | OpenAI Agents SDK | Vercel AI SDK |
|------|-----|------------|-----------|-------------------|---------------|
| 事件模型 | 异步 Promise 队列 | Hook 系统（18 事件） | 节点间通道 | 回调函数 | 回调函数 |
| 错误恢复 | auto-retry + overflow | Hook + 重试 | Checkpoint 回滚 | Guardrails | 无内置 |
| 权限模型 | Tool hooks | 5 步评估链 | 无内置 | Guardrails | 无内置 |
| 停止条件 | 自然 + abort + turn | turns + budget + hooks | 条件边 + END | handoff chain | stopWhen |
| 人机协作 | steering/follow-up | 5 种权限模式 | interrupt() + Command | 无内置 | prepareStep |

差异背后是不同的目标用户和设计哲学。Pi 追求极简可理解——一个开发者能读懂全部代码，所以用 Promise 链而非复杂的事件总线。Claude Code 面向企业——18 种 Hook 事件和 5 种权限模式是合规和审计的需要。LangGraph 来自学术背景——用图论的形式化保证取代工程上的"约定"。Vercel AI SDK 是库哲学——提供原语，不做假设，控制权交给用户。没有"最好"的控制回路设计，只有最适合你的约束条件的设计。

## 元技能：如何判断"这是控制问题"

当你的 Agent 出了问题，怎么判断它是控制回路的问题而不是其他层的问题？以下症状指向控制回路：

- **Agent 陷入循环**（重复相同操作）→ 稳定性问题，缺少循环检测或 turn 限制
- **Agent 不可停止**（Ctrl+C 是唯一选项）→ 缺少 abort 机制
- **事件丢失**（UI 不显示某些工具调用）→ 事件分发管道问题
- **错误级联**（一个 API 失败导致全部崩溃）→ 缺少隔离和重试
- **无法中途纠偏**（Agent 走偏了只能重启）→ 缺少 steering 消息队列

优先级判断：控制回路是第一个要建的 Harness 层——没有它，后续的工具、记忆、配置都无从挂载。一个没有控制回路的 Agent 无法可靠运行，其他层的设计再好也无法发挥作用。

衡量指标：错误恢复率（多少比例的瞬时错误被自动恢复）、平均恢复时间（从错误到恢复正常运行的耗时）、事件丢失率（事件流中有多少事件未被正确分发）。

## 简化思考

随着模型能力提升，某些重试逻辑可能变得不必要——模型自身能识别错误并调整策略。但有两类控制逻辑不会因模型进步而简化：

**生命周期管理**永远需要。只要 Agent 运行在外部环境中（操作系统、网络、文件系统），就需要启动序列、优雅关闭、资源清理。这是进程级的关切，与模型智能无关。

**事件分发**永远需要。持久化、UI 渲染、扩展通知——这些消费者需要从 Agent 获取信息。无论模型多聪明，信息路由是基础设施问题，不是智能问题。

简化方向：重试和降级策略可能从"硬编码规则"演变为"模型自适应"，但 Orchestrator 作为协调层的角色不会消失。
